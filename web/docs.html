SystemForge Documentation

SystemForge is a platform for learning system design by building real systems. You wire components together in a visual graph editor and write handler logic in SysLang — a simple scripting language.

How it works: Select a problem, connect components (servers, databases, caches), write handlers, and run tests. Your architecture is validated against test cases — just like LeetCode, but for distributed systems.

Quick Start

  1. Go to Problems and select Car Registration
  2. The graph shows: Client → HTTP Server → Database
  3. Design your schema — double-click the Database node to define tables and columns
  4. Click the server node to open the code editor
  5. Implement the handle_register function
  6. Click Run (or Ctrl+Enter) to test your solution

Your handler receives:

// req object passed to your handler:
{
    method: "POST",        // HTTP method
    path: "/register",     // request path
    body: {                // request body (map)
        plate: "ABC123",
        owner: "Alice"
    },
    params: {}             // URL parameters
}

Your handler returns:

// Return a map with status and body:
return {
    status: 201,
    body: { registered: true }
};

Variables

Use let to declare variables. No type annotations — types are inferred.

let name = "car_reg";
let count = 0;
let active = true;
let pi = 3.14;
let nothing = null;
let plates = ["ABC123", "XYZ789"];                      // list
let car = { plate: "ABC123", owner: "Alice", year: 2024 }; // map

Functions

Use fn to declare functions. Functions are first-class values and support closures.

fn validate_plate(plate) {
    if (plate.length() != 6) {
        return { ok: false, error: "invalid length" };
    }
    return { ok: true };
}

fn add(a, b) {
    return a + b;
}

// Functions can call other functions
fn process(plate) {
    let result = validate_plate(plate);
    if (!result.ok) {
        log("Invalid: " + result.error);
    }
    return result;
}

// Recursion works
fn factorial(n) {
    if (n <= 1) { return 1; }
    return n * factorial(n - 1);
}

Lambdas

Anonymous functions with C++-style capture syntax.

// No capture
let add = [](a, b) { return a + b; };
let result = add(3, 4);  // 7

// With capture (captured by value)
let threshold = 100;
let check = [threshold](val) { return val > threshold; };
check(150);  // true

// Closures
fn make_adder(x) {
    return [x](y) { return x + y; };
}
let add5 = make_adder(5);
add5(3);  // 8

Control Flow

// if / else if / else
if (count > 10) {
    log("too many");
} else if (count == 0) {
    log("none");
} else {
    log("ok");
}

// for — iterates over lists
for (let plate : plates) {
    log(plate);
}

// while
let i = 0;
while (i < 10) {
    log(i);
    i = i + 1;
}

// Logical operators
if (a > 0 && b > 0) { log("both positive"); }
if (a > 0 || b > 0) { log("at least one positive"); }
if (!active) { log("inactive"); }

Pattern Matching

Match expressions compare a value against patterns. Use _ as a wildcard.

fn handle_request(req) {
    return match (req.method) {
        "GET"  => get_data(req),
        "POST" => create_data(req),
        "PUT"  => update_data(req),
        _      => { status: 405, body: "Method not allowed" }
    };
}

Client Component

Represents users sending requests to your system. In tests, the client sends predefined requests and checks responses.

You don't write code for the Client. It's driven by test cases. Think of it as millions of users hitting your API.

HTTP Server Component

Handles incoming requests. You write handler functions that receive a request and return a response.

Request object

// Your handler receives:
fn my_handler(req) {
    req.method   // "GET", "POST", "PUT", "DELETE"
    req.path     // "/register", "/users"
    req.body     // map with request data
    req.params   // URL parameters
}

Response format

// Return a map with status + body:
return { status: 200, body: { data: "value" } };
return { status: 404, body: "Not found" };
return { status: 201, body: { created: true } };

Connected component access

When you connect a Database or Cache to your server in the graph, their APIs become available as variables:

// If server is connected to a Database (alias: "db"):
let car = db.FindOne("cars", { plate: "ABC123" });
db.Insert("cars", { plate: "ABC123", owner: "Alice" });

// If server is connected to a Cache (alias: "cache"):
let val = cache.get("key");
cache.set("key", "value", 3600);

Database Component

In-memory relational database with a functional API. No SQL — just simple function calls with maps.

MethodDescriptionReturns
db.FindOne(table, filter)Find first matching rowMap or null
db.Find(table, filter)Find all matching rowsList of maps
db.Insert(table, row)Insert a new row{ inserted: true }
db.Update(table, filter, updates)Update matching rows{ updated: N }
db.Delete(table, filter)Delete matching rows{ deleted: N }
db.All(table)Get all rowsList of maps
db.Count(table, filter)Count matching rowsNumber

The filter argument is a map of column-value pairs. All must match (AND logic). An empty filter {} matches all rows.

Cache Component

Key-value store with optional TTL (time-to-live in seconds).

MethodDescriptionReturns
cache.get(key)Get valueValue or null
cache.set(key, value, ttl)Set value (ttl optional)null
cache.del(key)Delete keytrue/false
cache.has(key)Check if key existstrue/false

Find / FindOne

// Find first matching row (or null if not found)
let car = db.FindOne("cars", { plate: "ABC123" });
// Returns: { plate: "ABC123", owner: "Alice" } or null

// Find all matching rows
let alice_cars = db.Find("cars", { owner: "Alice" });
// Returns: [ { plate: "ABC123", owner: "Alice" }, ... ]

// Empty filter matches all rows
let all = db.Find("cars", {});

Insert

db.Insert("cars", { plate: "ABC123", owner: "Alice" });
// Returns: { inserted: true }

// Insert with any columns your table has
db.Insert("users", { name: "Bob", email: "bob@test.com", age: 30 });

Update

// Update matching rows: (table, filter, updates)
db.Update("cars", { plate: "ABC123" }, { owner: "Bob" });
// Returns: { updated: 1 }

// Update multiple rows at once
db.Update("users", { role: "admin" }, { verified: true });
// Returns: { updated: N }

Delete

db.Delete("cars", { plate: "ABC123" });
// Returns: { deleted: 1 }

// Delete all matching rows
db.Delete("sessions", { expired: true });

All / Count

// Get all rows in a table
let cars = db.All("cars");
// Returns: [ { plate: "ABC123", owner: "Alice" }, ... ]

// Count rows matching a filter
let n = db.Count("cars", { owner: "Alice" });
// Returns: 2

// Count all rows (empty filter)
let total = db.Count("cars", {});
